set(User_SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/Src/main.c
    ${CMAKE_CURRENT_SOURCE_DIR}/Src/main_int.c
    ${CMAKE_CURRENT_SOURCE_DIR}/Src/motor.c
    ${CMAKE_CURRENT_SOURCE_DIR}/Src/protect.c

    ${User_SOURCES}
)

    ${CMAKE_CURRENT_SOURCE_DIR}/Inc
)

    INTERFACE User
)

    Core
    Middleware
    Communication
    Drivers
    Utils
)
